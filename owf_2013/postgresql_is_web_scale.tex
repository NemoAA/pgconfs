\frame{
  \frametitle{Strong Selling Points}

  \begin{itemize}
  \item Fully ACID: \textbf{trusted data}, \textbf{transactions}, \textbf{durability}
  \item Durability reads as expensive, reads also as \textbf{Crash Safety}
  \item Query planner and optimizer
  \item Advanced Indexing (BTree, GiST, GIN, SP-GiST)
  \item Procedure Languages (PL/C, PL/SQL, PL/pgSQL, PL/tcl, PL/perl, PL/python)
  \item Extensibility (data types, functions, operators, indexes)
  \end{itemize}
}
